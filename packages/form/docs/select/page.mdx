import { UiSelect } from '@uireact/form';

# UiSelect

> Component used to render select inputs

## Installation

> npm i @uireact/foundation @uireact/form

## Usage

```jsx live scope={{UiSelect}}
  <UiSelect
    onChange={(value) => {
      console.log(value);
    }}
    defaultValue=""
  >
    <option disabled value="">
      Select a car
    </option>
    <option> Ford </option>
    <option> Chevy </option>
  </UiSelect>
```

## UiSelect with selected value

```jsx live scope={{UiSelect}}
  <UiSelect label="Car" labelOnTop value="Chevy">
    <option disabled>Select a car</option>
    <option>Ford</option>
    <option>Chevy</option>
  </UiSelect>
```

## UiSelect disabled

```jsx live scope={{UiSelect}}
  <UiSelect label="Car" disabled>
    <option> Ford </option>
    <option> Chevy </option>
  </UiSelect>
```

## UiTextArea with error state and error message

```jsx live scope={{UiSelect}}
  <UiSelect label="Car" error="Some error" category="error">
    <option> Ford </option>
    <option> Chevy </option>
  </UiSelect>
```
