import { UiTextArea } from '@uireact/form';

# UiTextArea

> Component used to render text area fields

## Installation

> npm i @uireact/foundation @uireact/form

## Usage

```jsx live scope={{UiTextArea}}
  <UiTextArea
    placeholder="First Name"
    onChange={(value) => {
      console.log(value);
    }}
  />
```

### UiTextArea with label

```jsx live scope={{UiTextArea}}
  <UiTextArea label="Fist Name" labelOnTop cols={50} rows={3} />
```

### UiTextArea disabled

```jsx live scope={{UiTextArea}}
  <UiTextArea label="Fist Name" disabled />
```

### UiTextArea with state

```jsx live scope={{UiTextArea}}
  <UiTextArea label="Fist Name" category="positive" />
```

### UiTextArea with error state and error message

```jsx live scope={{UiTextArea}}
  <UiTextArea label="Fist Name" error="Some error" category="error" labelOnTop />
```
