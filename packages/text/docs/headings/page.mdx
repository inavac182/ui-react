import { UiHeading } from '@uireact/text';
import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

# UiHeading

<Metadata packageName='text' packageJson={packageJson} />

> Component to render texts

## Installation

> npm i -S @uireact/text

## Usage

```jsx live scope={{UiHeading}}
  <UiHeading level={1}>Some heading</UiHeading>
```

### UiHeading with inverse coloration

```jsx live scope={{UiHeading}}
  <UiHeading level={2} inverseColoration>
    Some heading
  </UiHeading>
```

### UiHeading centered

```jsx live scope={{UiHeading}}
  <UiHeading level={3} centered>
    Some heading
  </UiHeading>
```

### UiHeading wrapped

```jsx live scope={{UiHeading}}
  <UiHeading level={4} wrap>
    This is a very very very very very loooooong heading so it can be wrapped
  </UiHeading>
```

### UiHeading with default coloration

```jsx live scope={{UiHeading}}
<>
  <UiHeading level={5} coloration="dark">
    Will always render with DARK coloration
  </UiHeading>
  <UiHeading level={5} coloration="light">
    Will always render with LIGHT coloration
  </UiHeading>
</>
```

### UiHeading with spacing

```jsx live scope={{UiHeading}}
  <UiHeading level={6} padding={{all: 'five'}}>
    Some heading
  </UiHeading>
```
