import { UiButtonLink } from '@uireact/text';
import { UiIcon } from '@uireact/icons';
import { UiSpacing } from '@uireact/foundation';
import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

# UiButtonLink

<Metadata packageName='text' packageJson={packageJson} />

> Component to render a wrapper around links to provide styling, you have to provide the link element, this was done so we could support
> multiple use cases Link from react router, or Link from next JS or any other router you are using.

## Installation

> npm i -S @uireact/text

## Usage

```jsx live scope={{UiButtonLink}}
  <UiButtonLink>
    <a href="https://uireact.io/">Some text</a>
  </UiButtonLink>
```

### UiButtonLink - with options and spacing

```jsx live scope={{UiButtonLink, UiSpacing, UiIcon}}
  <UiButtonLink fullWidth inverseTextColoration size="xlarge" fontStyle="italic" padding={{ block: 'five' }}>
      <a href="https://uireact.io/">
        Some text <UiIcon icon="Link" inverseColoration />{' '}
      </a>
  </UiButtonLink>
```

### UiButtonLink with default coloration

```jsx live scope={{UiButtonLink}}
<>
  <UiButtonLink coloration="dark" margin={{ block: 'four' }}>
    <a href="https://uireact.io/">Will always render text as DARK coloration</a>
  </UiButtonLink>
  <UiButtonLink coloration="light">
    <a href="https://uireact.io/">Will always render text as LIGHT coloration</a>
  </UiButtonLink>
</>
```
