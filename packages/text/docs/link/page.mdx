import { UiLink } from '@uireact/text';
import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

import Link from 'next/link';

# UiLink

<Metadata packageName='text' packageJson={packageJson} />

> Component to render a wrapper around links to provide styling, you have to provide the link element, this was done so we could support
> multiple use cases Link from react router, or Link from next JS or any other router you are using.

## Installation

> npm i @uireact/foundation @uireact/text

## UiLink

```jsx live scope={{UiLink}}
  <UiLink>
    <a href="https://uireact.io/">Some text</a>
  </UiLink>
```

## UiLink with color category

```jsx live scope={{UiLink}}
<>
  <UiLink category="error">
    <a href="https://uireact.io/">Some text</a>
  </UiLink>{' '}
  <UiLink category="positive">
    <a href="https://uireact.io/">Some text</a>
  </UiLink>{' '}
  <UiLink handleClick={() => console.log('Link Clicked')} category="negative" fontStyle="italic">
    <a href="https://uireact.io/">Some text</a>
  </UiLink>
</>
```

## UiLink with default coloration

```jsx live scope={{UiLink}}
<>
  <UiLink coloration="dark">
    <a href="https://uireact.io/">Will always render with DARK coloration</a>
  </UiLink>
  <br />
  <UiLink coloration="light">
    <a href="https://uireact.io/">Will always render with LIGHT coloration</a>
  </UiLink>{' '}
</>
```

## UiLink with spacing

```jsx live scope={{UiLink}}
<>
  <UiLink padding={{all: 'five'}}>
    <a href="https://uireact.io/">Will always render with DARK coloration</a>
  </UiLink>
</>
```

## UiLink with NextJS Link

```jsx live scope={{UiLink, Link}}
<>
  <UiLink padding={{all: 'five'}}>
    <Link href="https://uireact.io/">Will always render with DARK coloration</Link>
  </UiLink>
</>
```
