import { UiExpandoText } from '@uireact/expando';

import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

# UiExpandoText
<Metadata packageName='expando' packageJson={packageJson} includeInformation description='This component renders a text and an icon that when expanded reveals some content underneath it.' />

## Usage

```jsx live scope={{UiExpandoText}}
  <UiExpandoText expandLabel="Expand me" collapseLabel="Collapse me">
    <span>This is some content</span>
  </UiExpandoText>
```

#### Expando with custom category and size

```jsx live scope={{UiExpandoText}}
  <UiExpandoText expandLabel="Expand me large" collapseLabel="Collapse me large" category="positive" size="xlarge">
    <span>This is some content</span>
  </UiExpandoText>
```

#### Expando initiated expanded

```jsx live scope={{UiExpandoText}}
  <UiExpandoText expandLabel="Expand me" collapseLabel="Collapse me" expanded>
    <span>This is some content</span>
  </UiExpandoText>
```
