import { UiExpandoCard } from '@uireact/expando';

import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

# UiExpandoCard
<Metadata packageName='expando' packageJson={packageJson} includeInformation description='This component renders a card with a heading that allows to show or hide content when clicked.' />

## Usage

```jsx live scope={{UiExpandoCard}}
  <UiExpandoCard expandLabel="Expand me" collapseLabel="Collapse me">
    <span>This is some content</span>
  </UiExpandoCard>
```

#### Expando with custom category and inverse heading coloration

```jsx live scope={{UiExpandoCard}}
  <UiExpandoCard expandLabel="Expand me" collapseLabel="Collapse me" category="tertiary" headingInverseColoration>
    <span>This is some content</span>
  </UiExpandoCard>
```

#### Expando initiated expanded

```jsx live scope={{UiExpandoCard}}
  <UiExpandoCard expandLabel="Expand me" collapseLabel="Collapse me" expanded>
    <span>This is some content</span>
  </UiExpandoCard>
```
