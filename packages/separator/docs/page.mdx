import { UiLineSeparator, UiDiagonalSeparator, UiBubblesSeparator } from '@uireact/separator';
import packageJson from './package-metadata.json';
import { Metadata } from '@uireact/docs-tools';

# UiSeparators

<Metadata packageName='separator' packageJson={packageJson} />

> A separator element to ease the change of colorations between blocks of content

## Installation

> npm i @uireact/foundation @uireact/separator

## UiLineSeparator

```jsx live scope={{UiLineSeparator}}
  <>
    <UiLineSeparator />
    <br />
    <UiLineSeparator inverse />
    <br />
    <UiLineSeparator delay={2000} />
  </>
```

## UiDiagonalSeparator

```jsx live scope={{UiDiagonalSeparator}}
  <UiDiagonalSeparator from="primary" to="secondary" />
```

## UiDiagonalSeparator inversed

```jsx live scope={{UiDiagonalSeparator}}
  <UiDiagonalSeparator from="primary" to="secondary" inverse />
```

## UiBubblesSeparator

```jsx live scope={{UiBubblesSeparator}}
  <UiBubblesSeparator from="secondary" to="primary" />
```

## UiBubblesSeparator inversed

```jsx live scope={{UiBubblesSeparator}}
  <UiBubblesSeparator from="secondary" to="primary" inverse />
```
