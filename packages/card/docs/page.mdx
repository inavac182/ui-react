import { UiText, UiHeading } from '@uireact/text';
import { UiSpacing, ColorTokens } from '@uireact/foundation';
import { UiCard } from '@uireact/card';

# UiCard

> Component to render content, it has built in styling and props to make it flexible.

## Installation

> npm i @uireact/foundation @uireact/card

## Usage

```jsx live scope={{UiCard, UiHeading, UiText}}
  <UiCard>
    <UiHeading>Card header</UiHeading>
    <UiText>This is the content</UiText>
    <UiText size="xsmall">This is the footer</UiText>
  </UiCard>
```

### Card as a link

```jsx live scope={{UiCard, UiText}}
  <UiCard link="https://www.uireact.io" bordered>
    <UiText>Open UiReact docs</UiText>
  </UiCard>
```

### UiCard with click handler

```jsx live scope={{UiCard, UiHeading}}
  <UiCard clickHandler={(identifier) => console.log(`Card ${identifier} clicked`)} identifier="My-ID" squared>
    <UiHeading>Click me!</UiHeading>
  </UiCard>
```

### UiCard with image content

```jsx live scope={{UiCard, UiHeading, UiText, UiSpacing}}
  <UiCard noPadding>
    <img
      src="https://upload.wikimedia.org/wikipedia/commons/e/e1/Colima_360.jpg"
      style={{ width: '100%', borderRadius: '3px 3px 0px 0px' }}
    />
    <UiSpacing padding={{ all: 'five' }}>
      <UiHeading>Card header</UiHeading>
      <UiText>This is the content</UiText>
      <UiText size="xsmall">This is the footer</UiText>
    </UiSpacing>
  </UiCard>
```

### UiCard outlined

```jsx live scope={{UiCard, UiHeading, UiText, UiSpacing}}
  <UiSpacing margin={{ block: 'three' }}>
    <UiCard weight="10" category="positive" styling="outlined">
      <h3>Card header</h3>
      <p>This is the content</p>
      <UiText size="small">This is the footer, using UiText styling is different.</UiText>
    </UiCard>
  </UiSpacing>
```

The outlined card will give some basic coloration to texts, although if you use any text component
the styling will be overwritten and has to be manually set up on each text component.
